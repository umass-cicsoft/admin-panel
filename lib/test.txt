StreamBuilder(
            stream: _database.child('members').orderByKey().onValue,
            builder: (context, snapshot) {
              if (snapshot.hasError) {
                return const Text("Error");
              }
              if (snapshot.connectionState == ConnectionState.waiting) {
                return const CircularProgressIndicator();
              }
              final data = snapshot.requireData;
              return SizedBox(
                  height: 500,
                  width: 300,
                  child: ListView.builder(
                      itemCount: data.size,
                      itemBuilder: (context, index) {
                        final _members = Member.fromJson(data.docs[index]);
                        //below is a slidable class that lets you slide to delete!
                        return ElevatedButton(
                            onPressed: () {
                              print(_members.name);
                            },
                            child: Text(_members.name));
                      }));
            },
          )
